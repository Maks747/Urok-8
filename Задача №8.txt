import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        Scanner scan = new Scanner(System.in);
        System.out.println ("¬ведите любое положительное число n");
        int n = scan.nextInt();
        int sum = 0;
        for (int i = 1; i <= n; i += 2) {
            sum += i;
            System.out.println(i);
        }
        System.out.println("—умма нечетных чисел  " + sum);
    }
}
